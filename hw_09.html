  day_count <- dim(apple)[1] - k
  day <- data.frame(y=tail(apple,day_count)$Open)
  for (j in 1:k){
    lst <- dim(apple)[1] - j
    day <-cbind(day,apple[(lst-day_count+1):(lst),])
  }
  colnames(day) <- c('y',cols[1:k])
  linearMod <- lm(y~., data=day)
  model_res[k]<- summary(linearMod)$r.squared
